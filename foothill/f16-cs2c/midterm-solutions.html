int search_in_sorted_array(
    const int ints[], const int x,
    const int low_index, const int high_index) {
  if (high_index - low_index <= 0) {
    return -1;
  }
  int middle_index = (low_index + high_index) / 2;
  if (ints[middle_index] < x) {
    return search_in_sorted_array(
        ints, x, middle_index + 1, high_index);
  } else if (ints[middle_index] > x) {
    return search_in_sorted_array(
        ints, x, low_index, middle_index);
  } else {
    return middle_index;
  }
}
O(log(high_index - low_index))

using namespace std;
void print_sorted_prefix_counts(
    const vector<string>& strs,
    const int prefix_len) {
  unordered_map<string, int> prefix_counts;
  cout << "Number of strings: "
            << strs.size() << endl;
  for (const string& str : strs) {
    ++prefix_counts[str.substr(0, prefix_len)];
  }
  vector<pair<string, int> > sorted_counts(
      prefix_counts.begin(), prefix_counts.end());
  sort(sorted_counts.begin(), sorted_counts.end());
  cout << "Number of unique prefixes: "
            << sorted_counts.size() << "\n";
  for (const auto& prefix_count : sorted_counts) {
    cout << prefix_count.first << ": "
              << prefix_count.second << endl;
  }
}
O(strs.size() + prefix_counts.size() * log(prefix_counts.size()))
